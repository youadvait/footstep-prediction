cmake_minimum_required(VERSION 3.15)
project(FootstepDetector-AU VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_BUILD_TYPE Release)

# Add JUCE subdirectory
add_subdirectory(JUCE)

# Create AU Plugin (same algorithm as VST2 for comparison)
juce_add_plugin(FootstepDetector-AU
    PRODUCT_NAME "FootstepDetector AU"
    COMPANY_NAME "FootstepTech"
    PLUGIN_MANUFACTURER_CODE FtTc
    PLUGIN_CODE FsDt
    FORMATS AU          # AudioUnit for Logic Pro
    IS_SYNTH FALSE
    NEEDS_MIDI_INPUT FALSE
    NEEDS_MIDI_OUTPUT FALSE
    IS_MIDI_EFFECT FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    AU_MAIN_TYPE kAudioUnitType_Effect
    COPY_PLUGIN_AFTER_BUILD TRUE
)

# Use same source files as VST2 (identical algorithm)
target_sources(FootstepDetector-AU PRIVATE
    vst_plugin/Source/PluginProcessor.cpp
    vst_plugin/Source/PluginEditor.cpp
    vst_plugin/Source/FootstepClassifier.cpp
)

# Same JUCE libraries
target_link_libraries(FootstepDetector-AU PRIVATE
    juce::juce_audio_basics
    juce::juce_audio_devices
    juce::juce_audio_formats
    juce::juce_audio_plugin_client
    juce::juce_audio_processors
    juce::juce_audio_utils
    juce::juce_core
    juce::juce_data_structures
    juce::juce_dsp
    juce::juce_events
    juce::juce_graphics
    juce::juce_gui_basics
    juce::juce_recommended_config_flags
    juce::juce_recommended_lto_flags
    juce::juce_recommended_warning_flags
)

# Same compile definitions as VST2
target_compile_definitions(FootstepDetector-AU PRIVATE
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_DISPLAY_SPLASH_SCREEN=0
    JUCE_REPORT_APP_USAGE=0
    JUCE_MODAL_LOOPS_PERMITTED=1
    # AU specific settings
    JucePlugin_Build_AU=1
    JucePlugin_Build_VST=0
    JucePlugin_Build_VST3=0
)
